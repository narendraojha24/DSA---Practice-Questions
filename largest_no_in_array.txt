#include <iostream>
using namespace std;

int main()
{
    int arr[]={1,2,3,4,5,89};
    int size = sizeof(arr)/sizeof(arr[0]);
    int max = 0;
    for(int i=0;i<size;i++){
        if(arr[i]>max){
            max = arr[i];
        }
    }
    
    cout<<"The largest number is : "<<max;

    return 0;
}
